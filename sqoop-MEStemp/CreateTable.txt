
CREATE DATABASE IF NOT EXISTS MEStemp;

DROP TABLE IF EXISTS MEStemp._SP_Crate;

Create External table IF NOT EXISTS MEStemp._SP_Crate (CrateID bigint NOT NULL, CrateNo nvarchar(508) NOT NULL, PartID int NOT NULL, Quantity float NOT NULL, CurrentQuantity float NOT NULL, Locked tinyint NOT NULL, Complete tinyint NOT NULL, CreatedDT datetime NOT NULL, CompletedDT datetime NOT NULL, MachineStageID smallint NOT NULL, CrateIndex bigint NOT NULL, MirrorPartID int NOT NULL, PadLock uniqueidentifier,  modcoID int NOT NULL) row format delimited fields terminated by ',' lines terminated by '\n' stored as textfile location '/MEStemp/_SP_Crate/';

DROP TABLE IF EXISTS MEStemp._SP_CrateLabel;

Create External table IF NOT EXISTS MEStemp._SP_CrateLabel (_SP_CrateLabelID bigint NOT NULL, CrateID bigint NOT NULL, MachineStage nvarchar(508), PartNumber nvarchar(508), InvState nvarchar(508), CrateNo nvarchar(508), CreatedDT datetime, CompleteDT datetime, Quantity float, LabelTemplate nvarchar(508), PrinterURL nvarchar(2048) NOT NULL, PrintedDT datetime NOT NULL, ScannedDT datetime NOT NULL, IsCancelled tinyint NOT NULL, IsReprint tinyint NOT NULL, PartDescription nvarchar(508), LabelSaveDirectory nvarchar(508), IsPrintLandscape tinyint NOT NULL, NoCopies smallint NOT NULL, LabelsPerInventoryPack smallint NOT NULL,  modcoID int NOT NULL) row format delimited fields terminated by ',' lines terminated by '\n' stored as textfile location '/MEStemp/_SP_CrateLabel/';
